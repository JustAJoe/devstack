#!/usr/bin/env bash
set +x


source ./stackrc


STACKSH_PARAMS=${STACKSH_PARAMS:-}


USE_CURRENT_DEVSTACK=${USE_CURRENT_DEVSTACK:-1}


RUN_SH=`mktemp`
cat > $RUN_SH <<EOF
#!/usr/bin/env bash
# DIVIDER
cd devstack
$STACKSH_PARAMS ./stack.sh
EOF


chmod 755 $RUN_SH

scp -r . root@$CONTAINER_IP:devstack
scp $RUN_SH root@$CONTAINER_IP:$RUN_SH
ssh root@$CONTAINER_IP $RUN_SH
